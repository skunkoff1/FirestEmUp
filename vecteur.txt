Classe Vecteur2D 
	
    /* ============== Attributs ===============================*/

	private var x : Reel
	private var y : Reel

    /* ================= Constructeurs =============================*/

	Public Fonction  Vecteur2D ( var x : entier, var y : entier) : NULL
		this.x = (Reel)x
		this.y = (Reel)y
	Finfonction

	Public Fonction Vecteur2D ( var x : reel, var y : reel) : NULL
		this.x = x
		this.y = y
	Finfonction
	
	Public Fonction Vecteur2D ( var autreVecteur : Vecteur2D) : NULL
		this.x = autreVecteur.getX()
		this.y = autreVecteur.getY()
	Finfonction

    /* ============= Getters et setters===================== */

	Public Fonction getX() : Entier ou Reel
            retourne this.x
    FinFonction

	Public Fonction getY() : Entier ou Reel
            retourne this.x
    FinFonction

	Public Fonction SetX(var nouveauxX : Reel) : NULL
            this.x = nouveauX
    FinFonction

	Public Fonction SetY(var nouveauxY : Reel) : NULL
            this.y = nouveauY
    FinFonction
